**[TextInput](/components/text-input) component with distinct set of possible values.**

## Import

```js transpile=false
import { Select } from '@byteclaw/visage';
```

## Examples

### Nonsearchable Select

```jsx live=true
() => {
  const [options] = useState(() =>
    Array(200)
      .fill(null)
      .map((_, i) => i),
  );

  return (
    <Select
      id="nonsearchable"
      onChange={value => console.log(value)}
      options={() => Promise.resolve(options)}
      placeholder="Select a value"
    />
  );
};
```

### Searchable Select

```jsx live=true
<Select
  id="searchable"
  onChange={value => console.log(value)}
  options={val =>
    Promise.resolve(
      Array(20)
        .fill(null)
        .map((v, i) => 'a'.repeat(i + 1))
        .filter(v => v.startsWith(val)),
    )
  }
  placeholder="Select a value"
  searchable
/>
```

### Disabled Select

```jsx live=true
<Select id="disabled-select" disabled placeholder="Select a value" />
```

### Invalid Select

```jsx live=true
<Select
  id="invalid-select"
  invalid
  searchable
  onChange={value => console.log(value)}
  options={filter =>
    new Promise(resolve =>
      resolve(
        Array(30)
          .fill(null)
          .map((i, id) => id)
          .filter(i => (filter ? i > Number(filter) : true))
          .map(i => i.toString()),
      ),
    )
  }
  placeholder="Select a value"
/>
```

### Limit menu height

By default there is no limit for `Menu` height so it&apos;ll take as much visible height as possible to fit `Menu` content.
You can limit this by creating your own menu using `createSelectMenu()` helper and passing it to `menu` prop.

```jsx live=true
() => {
  const limitedHeightMenu = useRef(
    createSelectMenu({
      defaultProps: {
        menuProps: {
          popoverProps: {
            styles: {
              maxHeight: 300,
            },
          },
        },
      },
    }),
  );
  const [options] = useState(() =>
    Array(100)
      .fill(null)
      .map((_, i) => i),
  );

  return (
    <Select
      id="nonsearchable"
      menu={limitedHeightMenu.current}
      onChange={value => console.log(value)}
      options={options}
      placeholder="Select a value"
    />
  );
};
```
