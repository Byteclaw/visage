**[Button](/component/button) like component that uses an icon.**

## Import

```js transpile=false
import { IconButton } from '@byteclaw/visage';
```

## Examples

### Default Button

```jsx live=true
<Fragment>
  <IconButton icon={MoonIcon} label="Default button" type="button" />
  <Box styles={{ mt: 2 }} />
  <IconButton icon={MoonIcon} label="Default button" stroked type="button" />
  <Box styles={{ mt: 2 }} />
  <IconButton disabled icon={MoonIcon} label="Default button" type="button" />
  <Box styles={{ mt: 2 }} />
  <IconButton
    icon={MoonIcon}
    label="Default button"
    styles={{ fontSize: 2, lineHeight: 2 }}
    type="button"
  />
</Fragment>
```

### Monochromatic Button

Monochromatic uses current color.

```jsx live=true

  <Box styles={{ color: 'tomato', mt: 2 }}>
    <IconButton
      icon={MoonIcon}
      label="Monochromatic button"
      monochromatic
      type="button"
    />
  </Box>
  <Box styles={{ color: 'tomato', mt: 2 }}>
    <IconButton
      icon={MoonIcon}
      label="Monochromatic button"
      monochromatic
      stroked
      type="button"
    />

      disabled
```
