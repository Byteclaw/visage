**A widget for multiline text input.**

## Import

```js transpile=false
import { TextArea } from '@byteclaw/visage';
```

## Examples

### Default

```jsx live=true
<TextArea placeholder="Placeholder" />
```

### ReadOnly

```jsx live=true
<TextArea defaultValue="ReadOnly TextArea" readOnly />
```

### Invalid

```jsx live=true
<TextArea invalid placeholder="Invalid TextArea" />
```

### Disabled

```jsx live=true
<TextArea disabled placeholder="Disabled TextArea" />
```

### Auto resizing

```jsx live=true
<TextArea autoResize placeholder="Auto resizing TextArea" />
```

### Controlled auto resizable TextArea

```jsx live=true
function ControlledTextArea() {
  const [value, setValue] = useState('');

  return (
    <TextArea
      autoResize
      onChange={e => setValue(e.currentTarget.value)}
      value={value}
    />
  );
}
```
