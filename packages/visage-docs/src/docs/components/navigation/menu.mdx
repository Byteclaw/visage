### Disabled events

`disableAutoFocusItem` props turns of auto-focusing the first item in the menu list.

```jsx live
function Example() {
  const anchorRef = useRef();
  const [open, setOpen] = useState(false);

  return (
    <Fragment>
      <Button
        aria-controls="buttons-menu"
        aria-haspopup="menu"
        aria-expanded={open}
        onClick={() => setOpen(!open)}
        ref={anchorRef}
      >
        {open ? 'Close' : 'Pop up'}
      </Button>
      <Menu
        disableAutoFocusItem
        anchor={anchorRef}
        id="buttons-menu"
        onClose={() => setOpen(false)}
        open={open}
      >
        <MenuItem>One</MenuItem>
        <MenuItem>Two</MenuItem>
        <MenuItem>Three</MenuItem>
        <Divider />
        <MenuItem>Button One</MenuItem>
        <MenuItem>Button Two</MenuItem>
        <MenuItem>Button Three</MenuItem>
      </Menu>
    </Fragment>
  );
}
```
