# Select

## Import

```js transpile=false
import { Select } from '@byteclaw/visage';
```

## Examples

### Nonsearchable Select

```jsx live=true
<form action="">
  <label htmlFor="nonsearchable">Select nonsearchable value</label>
  <WithState
    defaultValue={Array(200)
      .fill(null)
      .map((_, i) => i)}
  >
    {options => (
      <Select
        id="nonsearchable"
        onChange={value => console.log(value)}
        options={() => Promise.resolve(options)}
        placeholder="Select a value"
      />
    )}
  </WithState>
</form>
```

### Searchable Select

```jsx live=true
<form action="">
  <label htmlFor="searchable">Searchable select value</label>
  <Select
    id="searchable"
    onChange={value => console.log(value)}
    options={val =>
      Promise.resolve(
        Array(20)
          .fill(null)
          .map((v, i) => 'a'.repeat(i + 1))
          .filter(v => v.startsWith(val)),
      )
    }
    placeholder="Select a value"
    searchable
  />
</form>
```

### Disabled Select

```jsx live=true
<Select id="disabled-select" disabled placeholder="Select a value" />
```

### Invalid Select

```jsx live=true
<Select
  id="invalid-select"
  invalid
  searchable
  onChange={value => console.log(value)}
  options={filter =>
    new Promise(resolve =>
      resolve(
        Array(30)
          .fill(null)
          .map((i, id) => id)
          .filter(i => (filter ? i > Number(filter) : true))
          .map(i => i.toString()),
      ),
    )
  }
  placeholder="Select a value"
/>
```

### Static options

```jsx live=true
<WithState
  defaultValue={Array(100)
    .fill(null)
    .map((_, i) => i)}
>
  {options => (
    <Select
      id="nonsearchable"
      onChange={value => console.log(value)}
      options={options}
      placeholder="Select a value"
    />
  )}
</WithState>
```
