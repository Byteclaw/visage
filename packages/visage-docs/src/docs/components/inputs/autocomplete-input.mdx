# Autocomplete input

## Import

```js transpile=false
import { AutocompleteInput } from '@byteclaw/visage';
```

## Examples

### Default

```jsx live=true
function Example() {
  const options = useRef(
    Array(100)
      .fill(null)
      .map((_, i) => i),
  );

  return (
    <AutocompleteInput
      id="autocomplete-field"
      onChange={value => console.log(value)}
      options={filter => {
        return new Promise(resolve => {
          if (filter) {
            const regex = new RegExp(filter, 'i');

            return resolve(options.current.filter(o => regex.test(o)));
          }

          return resolve(options.current);
        });
      }}
      placeholder="Enter a value"
    />
  );
}
```

### Invalid

```jsx live=true
<AutocompleteInput
  id="autocomplete-invalid-field"
  invalid
  onChange={value => console.log(value)}
  options={() => Promise.resolve(['a', 'b', 'c', 'd'])}
/>
```
