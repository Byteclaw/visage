# Checkbox

## Import

```js transpile=false
import { Checkbox } from '@byteclaw/visage';
```

## Examples

### Default

```jsx live=true
<Fragment>
  <Checkbox defaultChecked label="Checkbox label" name="1" />
  <Checkbox label="Checkbox label" name="2" />
</Fragment>
```

### Hidden label

```jsx live=true
<Fragment>
  <Checkbox hiddenLabel label="Hidden label" name="3-1" />
  <Checkbox hiddenLabel label="Hidden label" name="3-2" />
</Fragment>
```

### Invalid

```jsx live=true
<Checkbox invalid label="Invalid checkbox" name="invalid-checkbox" />
```

### Read only

```jsx live=true
<Checkbox readOnly label="Read only checkbox" name="4" />
```

### Disabled

```jsx live=true
<Fragment>
  <Checkbox disabled label="Disabled checkbox" name="5" />
  <Checkbox disabled checked label="Disabled checked checkbox" name="6" />
</Fragment>
```

### Controlled

```jsx live=true
function Example() {
  const [value, setValue] = useState();

  return (
    <Checkbox
      checked={!!value}
      onChange={e => setValue(e.currentTarget.checked)}
      label="Controlled checkbox"
      name="6"
    />
  );
}
```
