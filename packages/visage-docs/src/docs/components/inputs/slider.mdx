# Slider

## Import

```js transpile=false
import { Slider } from '@byteclaw/visage';
```

## Examples

### Controlled

```jsx live=true
function Example() {
  const [value, setValue] = useState([10]);

  return (
    <Slider
      min={0}
      max={20}
      step={1}
      onChange={values => setValue(values)}
      onFinalChange={values => console.log('final', values)}
      values={value}
    />
  );
}
```

### Controlled with allowed steps

```jsx live=true
function Example() {
  const [value, setValue] = useState([10]);

  return (
    <Slider
      allowedValues={[0, 5, 10, 15]}
      min={0}
      max={20}
      step={5}
      onChange={values => setValue(values)}
      onFinalChange={values => console.log('final', values)}
      values={value}
    />
  );
}
```

### Disabled

```jsx live=true
function Example() {
  const [value, setValue] = useState([10]);

  return (
    <Slider
      allowedValues={[0, 5, 10]}
      disabled
      min={0}
      max={20}
      step={1}
      onChange={values => setValue(values)}
      values={value}
    />
  );
}
```

### Read only

```jsx live=true
function Example() {
  const [value, setValue] = useState([10]);

  return (
    <Slider
      min={0}
      max={20}
      onChange={values => setValue(values)}
      readOnly
      step={1}
      values={value}
    />
  );
}
```
