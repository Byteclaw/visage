# Slider

## Import

```js transpile=false
import { Slider } from '@byteclaw/visage';
```

### Controlled

```jsx live=true
function Example() {
  const [value, setValue] = useState([10]);

  return (
    <Slider
      min={0}
      max={20}
      step={1}
      onChange={values => setValue(values)}
      values={value}
    />
  );
}
```

### Controlled with allowed steps

```jsx live=true
function Example() {
  const [value, setValue] = useState([10]);

  return (
    <Slider
      allowedValues={[0, 5, 10]}
      min={0}
      max={20}
      step={1}
      onChange={values => setValue(values)}
      values={value}
    />
  );
}
```
