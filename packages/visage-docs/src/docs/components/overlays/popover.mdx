# Popover

## Import

```js transpile=false
import { Popover } from '@byteclaw/visage';
```

## Examples

```jsx live=true
function Example() {
  const anchorRef = useRef();
  const [open, setOpen] = useState(false);

  return (
    <Fragment>
      <Button onClick={() => setOpen(true)} ref={anchorRef} type="button">
        Show popover
      </Button>
      <Popover
        allowScrolling={false}
        alwaysVisible
        anchor={anchorRef}
        keepAnchorWidth
        fullscreen={[true, false]}
        open={open}
        onClose={() => setOpen(false)}
      >
        <Flex
          styles={{
            alignItems: 'center',
            backgroundColor: 'darkShadeOverlay',
            height: 500,
            width: '100%',
          }}
        >
          Content
        </Flex>
      </Popover>
    </Fragment>
  );
}
```
