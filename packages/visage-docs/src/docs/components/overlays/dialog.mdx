# Dialog

## Import

```js transpile=false
import { Dialog } from '@byteclaw/visage';
```

## Examples

```jsx live=true
function Example() {
  const [show, setShow] = useState(false);

  if (!show) {
    return (
      <Button onClick={() => setShow(true)} type="button">
        Show dialog
      </Button>
    );
  }

  return (
    <Dialog
      id="dialog-unique-id"
      label="Closable modal"
      onClose={e => {
        console.log(e);
        setShow(false);
      }}
    >
      {Array(50)
        .fill(null)
        .map((_, i) => (
          <form key={i} action="">
            <label htmlFor="searchable">Searchable select value</label>
            <Select
              onChange={value => console.log(value)}
              options={() => Promise.resolve(['a', 'b', 'c', 'd'])}
              placeholder="Select a value"
              searchable
            />
          </form>
        ))}
    </Dialog>
  );
}
```
